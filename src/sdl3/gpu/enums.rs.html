macro_rules! impl_with {
    (bitwise_and_or $x:ident $prim:ident) => {
        impl BitOr<$x> for $x {
            type Output = $x;
            fn bitor(self, rhs: $x) -> Self::Output {
                unsafe { std::mem::transmute((self as $prim) | (rhs as $prim)) }
            }
        }
        impl BitAnd<$x> for $x {
            type Output = $x;
            fn bitand(self, rhs: $x) -> Self::Output {
                unsafe { std::mem::transmute((self as $prim) & (rhs as $prim)) }
            }
        }
    };
}
